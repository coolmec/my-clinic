      rhesus: patientValue.rhesus,
      sexe: patientValue.sexe,
      allergies: patientValue.allergies,
      assurance: patientValue.assurance,

      pathologies: patientValue.pathologies,
      pays: patientValue.pays,
      pec: patientValue.pec,
      prenom: patientValue.prenom,
